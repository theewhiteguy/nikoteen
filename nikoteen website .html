import React, { useState } from "react";

import { BrowserRouter as Router, Route, Routes, Link } from "react-router-dom";

import { Card, CardContent } from "@/components/ui/card";

import { Button, Select } from "@/components/ui/button";

import { ShoppingCart, CreditCard, Trash } from "lucide-react";


const products = [

  { id: 1, name: "Classic T-Shirt", price: 29.99, image: "https://via.placeholder.com/150" },

  { id: 2, name: "Denim Jacket", price: 59.99, image: "https://via.placeholder.com/150" },

  { id: 3, name: "Slim Fit Jeans", price: 49.99, image: "https://via.placeholder.com/150" },

];


const LandingPage = () => (

  <div className="p-6 text-center bg-gray-100 min-h-screen flex flex-col justify-center items-center">

    <h1 className="text-5xl font-bold">Welcome to Nikoteen</h1>

    <p className="mt-4 text-lg text-gray-700">Premium Clothing for Every Occasion</p>

    <Link to="/products" className="mt-6 bg-blue-500 text-white px-6 py-3 rounded text-lg">Shop Now</Link>

  </div>

);


const Home = () => (

  <div className="p-6 text-center">

    <h1 className="text-3xl font-bold">Explore Our Clothing Collection</h1>

    <p className="mt-2">Trendy and Comfortable Apparel</p>

    <Link to="/products" className="mt-4 inline-block bg-blue-500 text-white px-4 py-2 rounded">Shop Now</Link>

  </div>

);


const ProductView = ({ addToCart }) => {

  const [selectedSize, setSelectedSize] = useState({});


  const handleSizeChange = (productId, size) => {

    setSelectedSize((prev) => ({ ...prev, [productId]: size }));

  };


  return (

    <div className="p-6 grid grid-cols-1 md:grid-cols-3 gap-4">

      {products.map((product) => (

        <Card key={product.id} className="p-4">

          <CardContent>

            <img src={product.image} alt={product.name} className="w-full" />

            <h2 className="text-xl font-semibold mt-2">{product.name}</h2>

            <p className="text-gray-600">${product.price}</p>

            <select

              className="mt-2 p-2 border rounded w-full"

              value={selectedSize[product.id] || ""}

              onChange={(e) => handleSizeChange(product.id, e.target.value)}

            >

              <option value="">Select Size</option>

              <option value="Small">Small</option>

              <option value="Medium">Medium</option>

              <option value="Large">Large</option>

              <option value="Extra Large">Extra Large</option>

            </select>

            <Button

              className="mt-2 w-full flex items-center justify-center"

              onClick={() => addToCart({ ...product, size: selectedSize[product.id] || "Unknown" })}

            >

              <ShoppingCart className="mr-2" /> Add to Cart

            </Button>

          </CardContent>

        </Card>

      ))}

    </div>

  );

};


const Cart = ({ cart, removeFromCart }) => (

  <div className="p-6">

    <h2 className="text-2xl font-bold">Shopping Cart</h2>

    {cart.length === 0 ? (

      <p>Your cart is empty.</p>

    ) : (

      <div>

        {cart.map((item, index) => (

          <div key={index} className="flex justify-between items-center border-b p-2">

            <p>{item.name} (Size: {item.size}) - ${item.price}</p>

            <Button className="text-red-500" onClick={() => removeFromCart(index)}>

              <Trash />

            </Button>

          </div>

        ))}

        <p className="mt-4 font-bold">Total: ${cart.reduce((total, item) => total + item.price, 0).toFixed(2)}</p>

        <Link to="/checkout" className="mt-4 inline-block bg-green-500 text-white px-4 py-2 rounded">Proceed to Checkout</Link>

      </div>

    )}

  </div>

);


const Contact = () => (

  <div className="p-6">

    <h2 className="text-2xl font-bold">Contact Us</h2>

    <p>Email: support@nikoteen.com</p>

    <p>Phone: +1 234 567 890</p>

  </div>

);


const Checkout = ({ cart }) => {

  const handlePayment = async () => {

    const response = await fetch("http://localhost:5000/api/checkout", {

      method: "POST",

      headers: { "Content-Type": "application/json" },

      body: JSON.stringify({ cart }),

    });

    const data = await response.json();

    alert(data.message);

  };


  return (

    <div className="p-6 text-center">

      <h2 className="text-2xl font-bold">Checkout</h2>

      <p>Total: ${cart.reduce((total, item) => total + item.price, 0).toFixed(2)}</p>

      <Button className="mt-2 flex items-center justify-center" onClick={handlePayment}>

        <CreditCard className="mr-2" /> Pay Now

      </Button>

    </div>

  );

};


const App = () => {

  const [cart, setCart] = useState([]);


  const addToCart = (product) => setCart([...cart, product]);

  const removeFromCart = (index) => setCart(cart.filter((_, i) => i !== index));


  return (

    <Router>

      <nav className="p-4 bg-gray-800 text-white flex justify-between">

        <Link to="/" className="font-bold">Nikoteen</Link>

        <Link to="/cart" className="flex items-center">

          <ShoppingCart className="mr-1" /> ({cart.length})

        </Link>

      </nav>

      <Routes>

        <Route path="/" element={<LandingPage />} />

        <Route path="/home" element={<Home />} />

        <Route path="/products" element={<ProductView addToCart={addToCart} />} />

        <Route path="/cart" element={<Cart cart={cart} removeFromCart={removeFromCart} />} />

        <Route path="/contact" element={<Contact />} />

        <Route path="/checkout" element={<Checkout cart={cart} />} />

      </Routes>

    </Router>

  );

};


export default App;
